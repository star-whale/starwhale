                    command == "_background_noise_"

                    or obj.size < 10240
                    or not path.name.endswith(".wav")

                    continue

                    self._auth,

                        mime_type=MIMEType.WAV,
                        shape=(1,),
                    ),
                )

                    "label": command,
                    "speaker_id": speaker_id,
                    "utterance_num": int(utterance_num),
                }
